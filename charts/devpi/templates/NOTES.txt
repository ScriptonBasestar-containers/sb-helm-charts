1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "devpi.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "devpi.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "devpi.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME {{ .Values.devpi.port }}:{{ .Values.devpi.port }}
  echo "Visit http://127.0.0.1:{{ .Values.devpi.port }}"
{{- end }}

2. Get the root password:
{{- if .Values.devpi.rootPassword }}
  echo Root Password: {{ .Values.devpi.rootPassword }}
{{- else if .Values.devpi.existingSecret }}
  echo Root Password: $(kubectl get secret --namespace {{ .Release.Namespace }} {{ .Values.devpi.existingSecret }} -o jsonpath="{.data.{{ .Values.devpi.secretKeyName }}}" | base64 -d)
{{- else }}
  NOTE: No root password configured. Set devpi.rootPassword in values.yaml
{{- end }}

3. Using devpi client:
  # Install client
  pip install devpi-client

  # Login
  devpi use http://{{ include "devpi.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local:{{ .Values.devpi.port }}
  devpi login root --password=YOUR_PASSWORD

  # Create user and index
  devpi user -c testuser password=test123
  devpi index -c dev bases=root/pypi

  # Upload packages
  devpi use testuser/dev
  devpi upload

4. Configure pip to use devpi:
  # Set index URL
  pip config set global.index-url http://{{ include "devpi.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local:{{ .Values.devpi.port }}/root/pypi/+simple/

  # Or use environment variable
  export PIP_INDEX_URL=http://{{ include "devpi.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local:{{ .Values.devpi.port }}/root/pypi/+simple/

{{- if .Values.postgresql.external.enabled }}
5. External PostgreSQL database:
  Host: {{ .Values.postgresql.external.host }}
  Database: {{ .Values.postgresql.external.database }}
  NOTE: Devpi will use PostgreSQL for metadata storage instead of SQLite
{{- else }}
5. Storage:
  Devpi is using SQLite for metadata storage (default)
  Consider using external PostgreSQL for production environments
{{- end }}
