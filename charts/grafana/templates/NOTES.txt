1. Get the Grafana admin password:

{{- if .Values.grafana.existingSecret }}
  Your Grafana password is stored in: {{ .Values.grafana.existingSecret }}
{{- else }}
  export GRAFANA_ADMIN_PASSWORD=$(kubectl get secret --namespace {{ .Release.Namespace }} {{ include "grafana.secretName" . }} -o jsonpath="{.data.admin-password}" | base64 -d)
{{- end }}

2. Access Grafana:

{{- if .Values.ingress.enabled }}
  Grafana URL: {{ include "grafana.externalUrl" . }}
{{- else }}
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "grafana.fullname" . }} 3000:{{ .Values.service.port }}
  Grafana URL: http://localhost:3000
{{- end }}

3. Login credentials:

  Username: {{ .Values.grafana.adminUser }}
  Password: $GRAFANA_ADMIN_PASSWORD

{{- else }}
  Type: SQLite (local file)

{{- end }}
