1. Get the Alertmanager URL by running these commands:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "alertmanager.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Alertmanager URL: http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "alertmanager.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "alertmanager.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Alertmanager URL: http://$SERVICE_IP:{{ .Values.service.port }}"
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "alertmanager.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:9093 to use Alertmanager"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 9093:9093
{{- end }}

2. Check the status of your Alertmanager deployment:
  kubectl get statefulset --namespace {{ .Release.Namespace }} {{ include "alertmanager.fullname" . }}
  kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "alertmanager.name" . }}"

{{- if gt (int .Values.replicaCount) 1 }}

3. Your Alertmanager is running in HA mode with {{ .Values.replicaCount }} replicas
  - Cluster peers will automatically discover each other
  - Check cluster status: kubectl exec -n {{ .Release.Namespace }} {{ include "alertmanager.fullname" . }}-0 -- wget -qO- http://localhost:9093/api/v2/status
{{- end }}

4. Common operations:
  # View configuration
  kubectl get configmap --namespace {{ .Release.Namespace }} {{ include "alertmanager.fullname" . }} -o yaml

  # Reload configuration (if web.enable-lifecycle is set)
  kubectl exec -n {{ .Release.Namespace }} {{ include "alertmanager.fullname" . }}-0 -- wget -qO- --post-data='' http://localhost:9093/-/reload

  # View alerts
  kubectl exec -n {{ .Release.Namespace }} {{ include "alertmanager.fullname" . }}-0 -- wget -qO- http://localhost:9093/api/v2/alerts

  # View silences
  kubectl exec -n {{ .Release.Namespace }} {{ include "alertmanager.fullname" . }}-0 -- wget -qO- http://localhost:9093/api/v2/silences

{{- if not .Values.persistence.enabled }}

WARNING: Persistence is disabled. Alert data will be lost when pods are restarted.
         Enable persistence by setting persistence.enabled=true
{{- end }}

{{- if .Values.serviceMonitor.enabled }}

5. ServiceMonitor is enabled. Prometheus Operator will automatically scrape metrics.
{{- end }}
